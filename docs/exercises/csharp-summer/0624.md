1.素数判断
```c#
Console.Write("请输入一个正整数：");
int number = int.Parse(Console.ReadLine());
bool isPrime = true;

if (number <= 1)
{
    isPrime = false;
}
else
{
    // 只需检查到平方根即可
    for (int i = 2; i <= Math.Sqrt(number); i++)
    {
        if (number % i == 0)
        {
            isPrime = false;
            break;
        }
    }
}

Console.WriteLine($"{number}是素数：{isPrime}");
```

2.九九乘法表
```c#
for (int i = 1; i <= 9; i++)
{
    for (int j = 1; j <= i; j++)
    {
        // 如果是特定乘积，改变输出颜色
        if (i * j == 12 || i * j == 24)
        {
            Console.ForegroundColor = ConsoleColor.Red;
            Console.Write($"{j}×{i}={i*j}\t");
            Console.ResetColor();
        }
        else
        {
            Console.Write($"{j}×{i}={i*j}\t");
        }
    }
    Console.WriteLine();
}
```
