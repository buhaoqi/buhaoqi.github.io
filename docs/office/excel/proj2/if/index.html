IF(logical_test, value_if_true, value_if_false)

еҸӮж•°иҜҙжҳҺвҖӢ

еҸӮж•°жҳҜеҗҰеҝ…йңҖиҜҙжҳҺ
logical_testеҝ…йңҖйҖ»иҫ‘еҲӨж–ӯиЎЁиҫҫејҸпјҲз»“жһңдёәTRUEжҲ–FALSEпјү
value_if_trueеҝ…йңҖжқЎд»¶дёәTRUEж—¶иҝ”еӣһзҡ„еҖј
value_if_falseеҝ…йңҖжқЎд»¶дёәFALSEж—¶иҝ”еӣһзҡ„еҖј

дәҢгҖҒеҹәзЎҖзӨәдҫӢвҖӢ

зӨәдҫӢж•°жҚ®иЎЁ1пјҡеӯҰз”ҹжҲҗз»©еҲӨж–ӯвҖӢ

ABCD
еӯҰеҸ·е§“еҗҚжҲҗз»©жҳҜеҗҰеҸҠж ј
S001еј дёү85
S002жқҺеӣӣ58
S003зҺӢдә”92
S004иөөе…ӯ45

еҹәзЎҖз”Ёжі•вҖӢ

  1. з®ҖеҚ•еҲӨж–ӯжҳҜеҗҰеҸҠж јпјҲ60еҲҶеҸҠж јпјү

    =IF(C2>=60, "еҸҠж ј", "дёҚеҸҠж ј")

  2. еӨҡзӯүзә§еҲӨж–ӯ

    =IF(C2>=90, "дјҳз§Җ",
        IF(C2>=80, "иүҜеҘҪ",
            IF(C2>=70, "дёӯзӯү",
                IF(C2>=60, "еҸҠж ј", "дёҚеҸҠж ј")
            )
        )
    )

  3. дёҺANDеҮҪж•°з»“еҗҲ

    =IF(AND(C2>=60, C2<=100), "жңүж•ҲжҲҗз»©", "ж— ж•ҲжҲҗз»©")

  4. дёҺORеҮҪж•°з»“еҗҲ

    =IF(OR(C2<0, C2>100), "ејӮеёёеҲҶж•°", "жӯЈеёёеҲҶж•°")

дёүгҖҒе®һйҷ…еә”з”ЁеңәжҷҜвҖӢ

еә”з”Ё1пјҡе‘ҳе·ҘиҖғеӢӨеҘ–йҮ‘и®Ўз®—вҖӢ

ABCD
е‘ҳе·ҘIDе§“еҗҚеҮәеӢӨеӨ©ж•°еҘ–йҮ‘
E001еј дёү22
E002жқҺеӣӣ18
E003зҺӢдә”25
E004иөөе…ӯ20
=IF(C2>=22, 500,
    IF(C2>=20, 300,
        IF(C2>=18, 100, 0)
    )
)

еә”з”Ё2пјҡй”Җе”®жҸҗжҲҗи®Ўз®—вҖӢ

ABCD
й”Җе”®е‘ҳй”Җе”®йўқжҸҗжҲҗжҜ”дҫӢжҸҗжҲҗйҮ‘йўқ
еј дёү150000
жқҺеӣӣ80000
зҺӢдә”250000
=IF(B2>200000, B2*0.1,
    IF(B2>100000, B2*0.08,
        IF(B2>50000, B2*0.05, B2*0.03)
    )
)

еӣӣгҖҒеөҢеҘ—IFеҮҪж•°вҖӢ

еөҢеҘ—иҜӯжі•вҖӢ

IF(жқЎд»¶1, з»“жһң1,
   IF(жқЎд»¶2, з»“жһң2,
      IF(жқЎд»¶3, з»“жһң3, й»ҳи®Өз»“жһң)
   )
)

зӨәдҫӢпјҡеӯҰз”ҹиҜ„зә§зі»з»ҹвҖӢ

=IF(C2>=95, "A+",
    IF(C2>=90, "A",
        IF(C2>=85, "B+",
            IF(C2>=80, "B",
                IF(C2>=75, "C+",
                    IF(C2>=70, "C",
                        IF(C2>=60, "D", "F")
                    )
                )
            )
        )
    )
)

жіЁж„ҸпјҡExcel 2016+зүҲжң¬ж”ҜжҢҒIFSеҮҪж•°з®ҖеҢ–еөҢеҘ—вҖӢ

=IFS(C2>=95, "A+",
     C2>=90, "A",
     C2>=85, "B+",
     C2>=80, "B",
     C2>=75, "C+",
     C2>=70, "C",
     C2>=60, "D",
     TRUE, "F")

дә”гҖҒIFеҮҪж•°дёҺе…¶д»–еҮҪж•°з»“еҗҲвҖӢ

1. дёҺSUMеҮҪж•°з»“еҗҲвҖӢ

=IF(SUM(A1:A10)>100, "иҫҫж Ү", "жңӘиҫҫж Ү")

2. дёҺCOUNTIFеҮҪж•°з»“еҗҲвҖӢ

=IF(COUNTIF(A1:A10, "е®ҢжҲҗ")=10, "е…ЁйғЁе®ҢжҲҗ", "жңүжңӘе®ҢжҲҗ")

3. дёҺVLOOKUPеҮҪж•°з»“еҗҲвҖӢ

=IF(ISERROR(VLOOKUP(A2, ж•°жҚ®иЎЁ!$A:$B, 2, FALSE)), "жңӘжүҫеҲ°", VLOOKUP(...))

4. дёҺTODAYеҮҪж•°з»“еҗҲвҖӢ

=IF(TODAY()>A2, "е·ІиҝҮжңҹ", "жңүж•ҲжңҹеҶ…")

е…ӯгҖҒIFеҮҪж•°иҝ”еӣһдёҚеҗҢзұ»еһӢеҖјвҖӢ

иҝ”еӣһж–Үжң¬вҖӢ

=IF(A1>10, "еӨ§дәҺ10", "е°ҸдәҺзӯүдәҺ10")

иҝ”еӣһж•°еӯ—вҖӢ

=IF(A1>10, 100, 50)

иҝ”еӣһе…¬ејҸвҖӢ

=IF(A1>10, B1*0.1, B1*0.05)

иҝ”еӣһеҚ•е…ғж јеј•з”ЁвҖӢ

=IF(A1>10, B1, C1)

иҝ”еӣһз©әеҖјвҖӢ

=IF(A1>10, "з¬ҰеҗҲ", "")
жҲ–
=IF(A1>10, "з¬ҰеҗҲ", NA())

дёғгҖҒеёёи§Ғй”ҷиҜҜеҸҠи§ЈеҶіж–№жі•вҖӢ

1. #NAME? й”ҷиҜҜвҖӢ

=if(A1>10, "жҳҜ", "еҗҰ")  вқҢ еҮҪж•°еҗҚеҝ…йЎ»еӨ§еҶҷ
=IF(A1>10, "жҳҜ", "еҗҰ")  вң…

2. еҸӮж•°ж•°йҮҸй”ҷиҜҜвҖӢ

=IF(A1>10, "жҳҜ")       вқҢ зјәе°‘з¬¬дёүдёӘеҸӮж•°
=IF(A1>10, "жҳҜ", "еҗҰ") вң…

3. еөҢеҘ—еұӮж•°иҝҮеӨҡвҖӢ

=IF(...IF(...)) жңҖеӨҡеөҢеҘ—64еұӮпјҲExcelдёҚеҗҢзүҲжң¬еҸҜиғҪжңүе·®ејӮпјү

4. ж–Үжң¬жҜ”иҫғй—®йўҳвҖӢ

=IF(A1="10", "жҳҜ", "еҗҰ")  вқҢ еҰӮжһңA1жҳҜж•°еӯ—10пјҢеҸҜиғҪдёҚеҢ№й…Қ
=IF(A1=10, "жҳҜ", "еҗҰ")   вң… жҲ– =IF(TEXT(A1,"0")="10", "жҳҜ", "еҗҰ")

е…«гҖҒз»јеҗҲз»ғд№ вҖӢ

з»ғд№ ж•°жҚ®иЎЁвҖӢ

ABCDEF
е‘ҳе·ҘIDе§“еҗҚйғЁй—Ёй”Җе”®йўқеҮәеӢӨзҺҮз»©ж•ҲиҜ„зә§
E001еј дёүй”Җе”®йғЁ15000095%
E002жқҺеӣӣжҠҖжңҜйғЁ098%
E003зҺӢдә”й”Җе”®йғЁ8000085%
E004иөөе…ӯй”Җе”®йғЁ25000090%
E005еӯҷдёғдәәдәӢйғЁ0100%

з»ғд№ йўҳвҖӢ

  1. ж №жҚ®й”Җе”®йўқеҲӨж–ӯжҳҜеҗҰиҫҫж ҮпјҲиҫҫж Үж ҮеҮҶпјҡ100000пјү

    =IF(D2>=100000, "иҫҫж Ү", "жңӘиҫҫж Ү")

  2. и®Ўз®—й”Җе”®жҸҗжҲҗпјҲи§„еҲҷпјҡ20дёҮд»ҘдёҠ10%пјҢ10-20дёҮ8%пјҢ5-10дёҮ5%пјҢ5дёҮд»ҘдёӢ3%пјү

    =IF(D2>200000, D2*0.1,
        IF(D2>100000, D2*0.08,
            IF(D2>50000, D2*0.05,
                IF(D2>0, D2*0.03, 0)
            )
        )
    )

  3. з»јеҗҲз»©ж•ҲиҜ„зә§пјҲй”Җе”®йўқиҫҫж Үдё”еҮәеӢӨзҺҮ>90%дёәAпјҢеҗҰеҲҷдёәBпјү

    =IF(AND(D2>=100000, E2>0.9), "A", "B")

  4. еҲӨж–ӯйғЁй—ЁжҳҜеҗҰй”Җе”®йғЁ

    =IF(C2="й”Җе”®йғЁ", "й”Җе”®дәәе‘ҳ", "йқһй”Җе”®дәәе‘ҳ")

  5. и®Ўз®—еҘ–йҮ‘пјҲй”Җе”®йғЁжҢүй”Җе”®йўқ5%пјҢе…¶д»–йғЁй—Ёеӣәе®ҡ2000пјү

    =IF(C2="й”Җе”®йғЁ", D2*0.05, 2000)

  6. еҲӨж–ӯжҳҜеҗҰжңүй”Җе”®дёҡз»©

    =IF(D2>0, "жңүдёҡз»©", IF(D2=0, "ж— дёҡз»©", "ж•°жҚ®й”ҷиҜҜ"))

  7. еҲӣе»әзҠ¶жҖҒж ҮзӯҫпјҲдјҳз§Җпјҡй”Җе”®йўқ>20дёҮдё”еҮәеӢӨзҺҮ>95%пјү

    =IF(AND(D2>200000, E2>0.95), "вӯҗдјҳз§Җе‘ҳе·Ҙ", "жҷ®йҖҡе‘ҳе·Ҙ")

д№қгҖҒиҝӣйҳ¶еә”з”ЁвҖӢ

1. ж•°з»„е…¬ејҸдёӯзҡ„IFвҖӢ

{=SUM(IF(A1:A10>10, B1:B10, 0))}  // и®Ўз®—еӨ§дәҺ10зҡ„еҜ№еә”еҖјд№Ӣе’Ң

2. IFеҮҪж•°дёҺжқЎд»¶ж јејҸз»“еҗҲвҖӢ

=IF(A1>TODAY(), "жңӘеҲ°жңҹ", "е·ІеҲ°жңҹ")  // еҸҜз”ЁдәҺжқЎд»¶ж јејҸи§„еҲҷ

3. еӨҡжқЎд»¶еҲӨж–ӯпјҲеӨҚжқӮйҖ»иҫ‘пјүвҖӢ

=IF(OR(AND(A1>10, B1<5), C1="жҳҜ"), "йҖҡиҝҮ", "дёҚйҖҡиҝҮ")

4. й”ҷиҜҜеӨ„зҗҶвҖӢ

=IFERROR(еҺҹе…¬ејҸ, "еҮәй”ҷж—¶жҳҫзӨәзҡ„еҶ…е®№")
=IF(ISERROR(еҺҹе…¬ејҸ), "еҮәй”ҷ", еҺҹе…¬ејҸ)

еҚҒгҖҒе®һжҲҳжҠҖе·§вҖӢ

жҠҖе·§1пјҡз®ҖеҢ–еөҢеҘ—IFвҖӢ

// дҪҝз”ЁLOOKUPд»ЈжӣҝеӨҡеұӮIF
=LOOKUP(A1, {0,60,70,80,90}, {"F","D","C","B","A"})

жҠҖе·§2пјҡIFеҮҪж•°+CHOOSEеҮҪж•°вҖӢ

=CHOOSE(MATCH(A1, {0,60,70,80,90}), "F","D","C","B","A")

жҠҖе·§3пјҡиҝ”еӣһеёҰж јејҸзҡ„еҶ…е®№вҖӢ

=IF(A1>10, "вң…иҫҫж Ү", "вқҢжңӘиҫҫж Ү")

жҠҖе·§4пјҡж•°жҚ®йӘҢиҜҒдёӯзҡ„IFвҖӢ

=IF($A$1="жҳҜ", $B$1:$B$10, $C$1:$C$10)  // еҠЁжҖҒдёӢжӢүеҲ—иЎЁ

еҚҒдёҖгҖҒжҖ§иғҪдјҳеҢ–е»әи®®вҖӢ

  1. йҒҝе…ҚиҝҮеәҰеөҢеҘ—пјҡи¶…иҝҮ7еұӮеөҢеҘ—иҖғиҷ‘дҪҝз”Ёе…¶д»–еҮҪж•°
  2. и®Ўз®—йЎәеәҸпјҡе°ҶжңҖеҸҜиғҪеҸ‘з”ҹзҡ„жғ…еҶөж”ҫеңЁеүҚйқў
  3. дҪҝз”ЁIFSеҮҪж•°пјҡExcel 2016+зүҲжң¬дҪҝз”ЁIFSжӣҙжё…жҷ°
  4. ж•°з»„е…¬ејҸжӣҝд»ЈпјҡеҜ№дәҺеӨҚжқӮжқЎд»¶еҲӨж–ӯпјҢиҖғиҷ‘дҪҝз”ЁFILTERеҮҪж•°
  5. зј“еӯҳи®Ўз®—з»“жһңпјҡеҜ№дәҺйҮҚеӨҚдҪҝз”Ёзҡ„жқЎд»¶пјҢе…Ҳи®Ўз®—е№¶еӯҳеӮЁеҲ°иҫ…еҠ©еҲ—

еҚҒдәҢгҖҒжҖ»з»“вҖӢ

IFеҮҪж•°жҳҜExcelдёӯжңҖйҮҚиҰҒгҖҒжңҖеёёз”Ёзҡ„йҖ»иҫ‘еҮҪж•°д№ӢдёҖпјҢжҺҢжҸЎе®ғзҡ„дҪҝз”ЁжҠҖе·§еҸҜд»Ҙпјҡ

  1. е®һзҺ°ж•°жҚ®иҮӘеҠЁеҲҶзұ»
  2. еҲӣе»әеҠЁжҖҒи®Ўз®—е…¬ејҸ
  3. жһ„е»әжҷәиғҪжҠҘиЎЁ
  4. е®һзҺ°ж•°жҚ®йӘҢиҜҒе’Ңжё…жҙ—
  5. жҸҗеҚҮе·ҘдҪңж•ҲзҺҮе’ҢеҮҶзЎ®жҖ§
