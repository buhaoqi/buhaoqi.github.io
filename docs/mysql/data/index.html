教师表:teacher​

-- 创建教师表 teacher
CREATE TABLE IF NOT EXISTS teacher (
    TeaID INT UNSIGNED NOT NULL COMMENT '教师工号（主键）',
    Teaname VARCHAR(20) NOT NULL COMMENT '教师姓名',
    Age TINYINT UNSIGNED NOT NULL COMMENT '教师年龄（1-255范围，符合实际）',
    -- 工号作为唯一标识，设置为主键
    PRIMARY KEY (TeaID)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='教师信息表（用于视图示例）';

用户表: users​

-- 创建数据库（如果不存在）
CREATE DATABASE IF NOT EXISTS demo_db;
USE demo_db;

-- 创建 users 表
CREATE TABLE users (
    id INT PRIMARY KEY AUTO_INCREMENT,
    username VARCHAR(50) UNIQUE NOT NULL,
    email VARCHAR(100) UNIQUE NOT NULL,
    password_hash VARCHAR(255) NOT NULL,
    full_name VARCHAR(100) NOT NULL,
    age INT,
    gender ENUM('男', '女', '其他') DEFAULT '其他',
    country VARCHAR(50) DEFAULT '中国',
    city VARCHAR(50),
    registration_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    last_login TIMESTAMP NULL,
    is_active BOOLEAN DEFAULT TRUE,
    account_balance DECIMAL(10, 2) DEFAULT 0.00,
    membership_level ENUM('普通', '白银', '黄金', '铂金', '钻石') DEFAULT '普通',
    INDEX idx_username (username),
    INDEX idx_email (email),
    INDEX idx_registration_date (registration_date),
    INDEX idx_is_active (is_active)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

学生成绩​

建表

-- 创建学生表 students
CREATE TABLE IF NOT EXISTS students (
    -- 新增自增主键 id，用于唯一标识每条学生记录（INSERT 中未提供，建表建议必备）
    s_id INT UNSIGNED NOT NULL AUTO_INCREMENT COMMENT '学生唯一标识ID',
    -- 学生姓名
    s_name VARCHAR(20) NOT NULL COMMENT '学生姓名',
    -- 出生日期（匹配 INSERT 中的 'YYYY-MM-DD' 格式）
    s_birthdate DATE NOT NULL COMMENT '学生出生日期',
    -- 性别（仅男/女，固定长度字符串更节省空间）
    s_gender CHAR(2) NOT NULL COMMENT '学生性别（男/女）',
    -- 籍贯（区县名称，长度足够覆盖场景）
    s_hometown VARCHAR(20) NOT NULL COMMENT '学生籍贯（区县）',
    -- 各科成绩（保留2位小数，总长度5支持 999.99 的成绩范围，满足需求）
    s_chinese DECIMAL(5, 2) NOT NULL COMMENT '语文成绩',
    s_math DECIMAL(5, 2) NOT NULL COMMENT '数学成绩',
    s_english DECIMAL(5, 2) NOT NULL COMMENT '英语成绩',
    s_computer DECIMAL(5, 2) NOT NULL COMMENT '计算机成绩',
    -- 是否有特长（仅是/否，固定长度字符串）
    s_specialty CHAR(2) NOT NULL COMMENT '是否有特长（是/否）',
