## 找出字符串中所有的数字charCodeAt()

```csharp
var str = 'haj123sdk54hask33dkhalsd879';
function findNum(str){
	var arr = [];
	var tmp = '';
	for(var i=0;i<str.length;i++){
		if( str.charAt(i)<='9' && str.charAt(i)>='0' ){
			tmp += str.charAt(i);
		}
		else{
			if(tmp){
				arr.push(tmp);
				tmp = '';
			}
		}
	}
	
	if(tmp){
		arr.push(tmp);
		tmp = '';
	}
	
	return arr;
	
}
```

## 出现次数最多的字符

判断一个字符串 'abcoefoxyozzopp' 中出现次数最多的字符，并统计其次数。

```csharp
// 有一个对象 来判断是否有该属性 对象['属性名']
var o = {
  age: 18
};
if (o["sex"]) {
  console.log("里面有该属性");
} else {
  console.log("没有该属性");
}

//  判断一个字符串 'abcoefoxyozzopp' 中出现次数最多的字符，并统计其次数。
// o.a = 1
// o.b = 1
// o.c = 1
// o.o = 4
// 核心算法：利用 charAt(） 遍历这个字符串
// 把每个字符都存储给对象， 如果对象没有该属性，就为1，如果存在了就 +1
// 遍历对象，得到最大值和该字符
var str = "abcoefoxyozzopp";
var o = {};
for (var i = 0; i < str.length; i++) {
  var chars = str.charAt(i); // chars 是 字符串的每一个字符
  if (o[chars]) {
    // o[chars] 得到的是属性值
    o[chars]++;
  } else {
    o[chars] = 1;
  }
}
console.log(o);
// 2. 遍历对象
var max = 0;
var ch = "";
for (var k in o) {
  // k 得到是 属性名
  // o[k] 得到的是属性值
  if (o[k] > max) {
    max = o[k];
    ch = k;
  }
}
console.log(max);
console.log("最多的字符是" + ch);


```
