```csharp
一、输出 1 2 3
for (int i = 1; i <= 3; i++)
{
    Console.WriteLine(i); //1,2,3
}
二、输出 3,6,9
for (int i = 1; i <= 3; i++)
{
    Console.WriteLine(i * 3); //3,6,9
}
三、输出1 ～ 100之间的偶数
for (int i = 2; i <= 100; i = i + 2)
{
    Console.WriteLine(i); 
}
四、输出1 ～ 100之间的奇数
for (int i = 1; i <= 99; i = i + 2)
{
    Console.WriteLine(i); 
}

五、输出 3,2,1
for (int i = 3; i >= 1; i = i - 1)
{
    Console.WriteLine(i); 
}
六、计算 1~100之间的和
设和是 x
int x = 0;
for (int i = 1; i<=100; i = i+1)
{
    x = x + i; 
    Console.WriteLine(i);
}
Console.WriteLine(x);
七、输出 1～100之间偶数的和
int x = 0;
for (int i = 2; i<=100; i = i+2)
{
    x = x + i; 
    Console.WriteLine(i);
}
Console.WriteLine(x);
// 八、 输出1 ~ 100之间能被3整除不能被5整除的数
for (int i = 1; i <= 100; i++)
{
    if (i % 3 == 0 && i % 5 != 0)
    {
        Console.WriteLine(i);
    }
}
```
