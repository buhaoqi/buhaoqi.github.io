## 猜字母游戏

```csharp
 // 游戏标题
 Console.WriteLine("==================================");
 Console.WriteLine("    🎮 文字猜谜大挑战 🎮");
 Console.WriteLine("==================================");
 Console.WriteLine("提示：输入一个字母来猜，猜对会显示，错了扣生命值～\n");

 // 随机单词库
 string[] words = { "apple", "banana", "computer", "student", "family", "hello", "unity" };
 Random random = new Random();
 string answer = words[random.Next(words.Length)]; // 随机选一个答案

 // 演示字符串技能：Length、Replace、ToUpper
 Console.WriteLine($"【技能演示】单词长度：{answer.Length} 个字母");
 string demoReplace = answer.Replace("a", "@");
 Console.WriteLine($"【技能演示】将单词中的 'a' 替换为 '@'：{demoReplace}");
 string demoUpper = answer.ToUpper();
 Console.WriteLine($"【技能演示】单词大写形式：{demoUpper}\n");

 // 生成隐藏文字：首字母显示，其余变 *
 char[] hidden = answer.ToCharArray();
 for (int i = 1; i < hidden.Length; i++)
 {
     hidden[i] = '*';
 }
 string hiddenWord = new string(hidden);

 int life = 5;                     // 生命值
 List<char> guessedLetters = new List<char>(); // 记录已猜过的字母（大写存储，便于比较）

 // 游戏主循环
 while (life > 0 && hiddenWord.Contains("*"))
 {
     Console.WriteLine($"\n当前单词：{string.Join(" ", hiddenWord.ToCharArray())}"); // 空格分隔更清晰
     Console.WriteLine($"剩余生命值：{life}");
     Console.Write("请输入一个字母：");

     string input = Console.ReadLine();
     // 输入验证：不能为空，且必须是单个字母
     if (string.IsNullOrEmpty(input) || input.Length != 1 || !char.IsLetter(input[0]))
     {
         Console.WriteLine("❌ 无效输入！请输入一个字母（a-z 或 A-Z）。");
         continue;
     }

     char guess = char.ToUpper(input[0]);   // 统一转大写存储，方便去重比较
     char guessLower = char.ToLower(guess); // 小写用于与单词比较

     // 检查是否已经猜过这个字母
     if (guessedLetters.Contains(guess))
     {
         Console.WriteLine($"⚠️ 字母 '{guess}' 已经猜过了，请换一个！");
         continue;
     }
     guessedLetters.Add(guess);

     // 使用 Contains 判断字母是否在单词中（技能演示）
     bool isCorrect = answer.Contains(guessLower);
     // 也可用 IndexOf 演示（技能演示）
     int index = answer.IndexOf(guessLower);
     if (index >= 0)
     {
         // 仅演示 IndexOf 的用法，不额外处理
         // Console.WriteLine($"调试：字母首次出现在位置 {index}");
     }

     if (isCorrect)
     {
         Console.WriteLine("✅ 恭喜！猜对了！");

         // 把所有猜对的字母显示出来（使用 ToCharArray 技能）
         char[] temp = hiddenWord.ToCharArray();
         for (int i = 0; i < answer.Length; i++)
         {
             if (answer[i] == guessLower)
             {
                 temp[i] = guessLower; // 显示小写字母
             }
         }
         hiddenWord = new string(temp);
     }
     else
     {
         life--;
         Console.WriteLine($"❌ 猜错啦！生命值-1，剩余：{life}");
     }
 }

 // 游戏结束判断
 if (life == 0)
 {
     Console.WriteLine("\n💀 游戏结束！正确答案是：" + answer);
 }
 else
 {
     Console.WriteLine("\n🎉 恭喜你赢了！正确答案：" + answer);
 }

 Console.ReadLine(); // 防止控制台立即关闭
 ```
