    12.ж•ҙеһӢе’Ңжө®зӮ№еһӢеёёйҮҸ

    йқһеёёеҘҪпјҒдҪ й—®еҲ°дәҶ C# дёӯдёӨдёӘйқһеёёйҮҚиҰҒзҡ„ж•ҙеһӢеёёйҮҸпјҡ

    вң… int.MaxValue е’Ң int.MinValue

    е®ғд»¬жҳҜ C# дёӯ int зұ»еһӢпјҲеҚі 32 дҪҚжңүз¬ҰеҸ·ж•ҙж•°пјҢSystem.Int32пјүзҡ„дёӨдёӘйқҷжҖҒеҸӘиҜ»еӯ—ж®өпјҲеёёйҮҸпјүпјҢз”ЁдәҺиЎЁзӨәпјҡ

    • int.MaxValue вҶ’ int зұ»еһӢиғҪеӯҳеӮЁзҡ„жңҖеӨ§еҖј
    • int.MinValue вҶ’ int зұ»еһӢиғҪеӯҳеӮЁзҡ„жңҖе°ҸеҖј

    дёҖгҖҒрҹ”ў еҹәжң¬е®ҡд№үдёҺеҖј

    еёёйҮҸ жүҖеұһзұ»еһӢ еҖјпјҲеҚҒиҝӣеҲ¶пјү иҜҙжҳҺ
    int.MaxValue intпјҲеҚі System.Int32пјү 2,147,483,647пјҲеҚі 2ВіВ№ - 1пјү int иғҪиЎЁзӨәзҡ„жңҖеӨ§жӯЈж•ҙж•°
    int.MinValue intпјҲеҚі System.Int32пјү -2,147,483,648пјҲеҚі -2ВіВ№пјү int иғҪиЎЁзӨәзҡ„жңҖе°Ҹиҙҹж•ҙж•°

    вҡ пёҸ жіЁж„Ҹпјҡ - int жҳҜжңүз¬ҰеҸ·зҡ„ 32 дҪҚж•ҙж•°пјҢиҢғеӣҙдёәпјҡ-2,147,483,648 пҪһ 2,147,483,647 - е®ғд»¬жҳҜ йқҷжҖҒеӯ—ж®өпјҲдёҚжҳҜж–№жі•пјүпјҢжүҖд»ҘжІЎжңүжӢ¬еҸ·пјҢжӯЈзЎ®еҶҷжі•жҳҜ int.MaxValueпјҢдёҚжҳҜ int.MaxValue()

    дәҢгҖҒрҹ“Ң дёәд»Җд№ҲйңҖиҰҒ int.MaxValue е’Ң int.MinValueпјҹ

    еңЁе®һйҷ…зј–зЁӢдёӯпјҢжҲ‘д»¬з»ҸеёёйңҖиҰҒпјҡ

    1. дәҶи§Јж•°жҚ®зұ»еһӢзҡ„иҫ№з•ҢпјҢйҳІжӯўж•°еҖјжәўеҮәпјҲOverflowпјү
    2. еҲқе§ӢеҢ–жңҖеӨ§еҖј/жңҖе°ҸеҖјеҸҳйҮҸпјҢжҜ”еҰӮеңЁжүҫж•°з»„зҡ„жңҖеӨ§/жңҖе°Ҹе…ғзҙ ж—¶
    3. еҒҡиҫ“е…Ҙж ЎйӘҢпјҢзЎ®дҝқз”ЁжҲ·иҫ“е…Ҙзҡ„ж•°еӯ—еңЁеҗҲзҗҶиҢғеӣҙеҶ…
    4. иҝӣиЎҢж•°еҖјжҜ”иҫғгҖҒиҢғеӣҙйҷҗеҲ¶гҖҒе®үе…Ёи®Ўз®—

    дёүгҖҒрҹ”Қ еёёйҮҸиҜҰи§ЈдёҺз”Ёжі•

    вң… 1. и®ҝй—®ж–№ејҸ

    е®ғд»¬жҳҜ int зұ»еһӢзҡ„йқҷжҖҒеӯ—ж®өпјҢзӣҙжҺҘйҖҡиҝҮзұ»еһӢеҗҚи®ҝй—®пјҡ

    Console.WriteLine("int жңҖеӨ§еҖј: " + int.MaxValue);   // иҫ“еҮәпјҡ2147483647
Console.WriteLine("int жңҖе°ҸеҖј: " + int.MinValue);   // иҫ“еҮәпјҡ-2147483648

    вқҢ й”ҷиҜҜеҶҷжі•пјҡint.MaxValue() вҶ’ зј–иҜ‘жҠҘй”ҷпјҢеӣ дёәе®ғд»¬дёҚжҳҜж–№жі•пјҒ

    вң… 2. е…ёеһӢдҪҝз”ЁеңәжҷҜдёҺзӨәдҫӢд»Јз Ғ

    рҹҺҜ еңәжҷҜ 1пјҡйҳІжӯўж•°еҖјжәўеҮәпјҲOverflowпјү

    еҪ“иҝӣиЎҢж•°еӯҰиҝҗз®—ж—¶пјҢжҜ”еҰӮзҙҜеҠ гҖҒд№ҳжі•пјҢз»“жһңеҸҜиғҪдјҡи¶…еҮә int зҡ„иҢғеӣҙпјҢеҜјиҮҙж•°жҚ®жәўеҮәпјҢз»“жһңеҸҜиғҪеӣһз»•пјҲеҸҳжҲҗиҙҹж•°жҲ–е…¶д»–й”ҷиҜҜеҖјпјүгҖӮ

    вң… дҪҝз”Ё int.MaxValue е’Ң int.MinValue еҸҜд»ҘжҸҗеүҚеҲӨж–ӯпјҢйҒҝе…ҚжәўеҮәйЈҺйҷ©гҖӮ

    рҹ§Ә зӨәдҫӢпјҡжЈҖжөӢеҠ жі•жҳҜеҗҰеҸҜиғҪжәўеҮә
    int a = 2147483640;
int b = 10;

if (a > int.MaxValue - b)
{
    Console.WriteLine("й”ҷиҜҜпјҡеҠ жі•иҝҗз®—дјҡеҜјиҮҙжәўеҮәпјҒ");
}
else
{
    int sum = a + b;
    Console.WriteLine("е’Ңдёәпјҡ" + sum); // е®үе…Ёж—¶жүҚи®Ўз®—
}

    рҹҺҜ и§ЈйҮҠпјҡеҰӮжһң a + b > int.MaxValueпјҢе°ұдјҡжәўеҮәгҖӮжҲ‘д»¬йҖҡиҝҮ int.MaxValue - b жҸҗеүҚеҲӨж–ӯгҖӮ

    рҹҺҜ еңәжҷҜ 2пјҡеҲқе§ӢеҢ–жңҖеӨ§еҖје’ҢжңҖе°ҸеҖјпјҲжҜ”еҰӮжүҫжңҖеҖјпјү

    еңЁйҒҚеҺҶж•°з»„жҲ–йӣҶеҗҲпјҢжҹҘжүҫ жңҖеӨ§еҖј жҲ– жңҖе°ҸеҖј ж—¶пјҢйҖҡеёёдјҡиҝҷж ·еҲқе§ӢеҢ–пјҡ

    int[] numbers = { 10, 5, 20, 15, 30 };

// еҲқе§ӢеҢ–жңҖеӨ§еҖјдёә int зұ»еһӢжңҖе°ҸеҸҜиғҪеҖј
int max = int.MinValue;

// еҲқе§ӢеҢ–жңҖе°ҸеҖјдёә int зұ»еһӢжңҖеӨ§еҸҜиғҪеҖј
int min = int.MaxValue;

foreach (int num in numbers)
{
    if (num > max)
        max = num;

    if (num < min)
        min = num;
}

Console.WriteLine("жңҖеӨ§еҖј: " + max); // 30
Console.WriteLine("жңҖе°ҸеҖј: " + min); // 5

    вң… иҝҷжҳҜдёҖз§Қйқһеёёеёёи§Ғдё”е®үе…Ёзҡ„еҲқе§ӢеҢ–ж–№ејҸпјҒ

    рҹҺҜ еңәжҷҜ 3пјҡз”ЁжҲ·иҫ“е…Ҙж ЎйӘҢпјҲиҢғеӣҙйҷҗеҲ¶пјү

    еҒҮи®ҫдҪ еёҢжңӣз”ЁжҲ·иҫ“е…ҘдёҖдёӘе№ҙйҫ„пјҢеҗҲзҗҶзҡ„иҢғеӣҙжҳҜ 0пҪһ120пјҢдҪ д№ҹеҸҜд»Ҙе’Ң int зҡ„иҢғеӣҙеҒҡеҜ№жҜ”пјҲиҷҪз„¶жһҒе°‘йңҖиҰҒзӣҙжҺҘе’Ң int.MaxValue жҜ”иҫғпјҢдҪҶйҖ»иҫ‘зұ»дјјпјүгҖӮ

    Console.WriteLine("иҜ·иҫ“е…Ҙе№ҙйҫ„пјҡ");
if (int.TryParse(Console.ReadLine(), out int age))

    рҹ§  иҷҪз„¶е№ҙйҫ„дёҚдјҡжҺҘиҝ‘ int.MaxValueпјҢдҪҶзұ»дјјең°пјҢдҪ еҸҜд»Ҙз”ЁиҝҷдәӣеёёйҮҸжқҘйҷҗе®ҡиҫ“е…ҘиҢғеӣҙпјҢжҲ–еңЁеӨ„зҗҶжӣҙеӨ§ж•°еӯ—пјҲеҰӮи®ўеҚ• IDгҖҒдәӨжҳ“йҮҸпјүж—¶еҒҡдҝқжҠӨгҖӮ

    рҹҺҜ еңәжҷҜ 4пјҡж–ҮжЎЈиҜҙжҳҺ / еёёйҮҸе®ҡд№үеҸӮиҖғ

    еңЁе®ҡд№үиҮӘе·ұзҡ„еёёйҮҸж—¶пјҲжҜ”еҰӮжңҖеӨ§з”ЁжҲ·ж•°гҖҒжңҖеӨ§йЎөз ҒгҖҒзј“еҶІеҢәеӨ§е°ҸзӯүпјүпјҢеҸҜд»ҘеҸӮиҖғ int.MaxValue дҪңдёәзҗҶи®әдёҠйҷҗгҖӮ

    // еҒҮи®ҫжҲ‘д»¬е®ҡд№үжңҖеӨ§е…Ғи®ёзҡ„з”ЁжҲ·ж•°йҮҸ
const int MaxUsers = 1000000; // иҮӘе·ұе®ҡд№үзҡ„дёҠйҷҗпјҢйҖҡеёёиҝңе°ҸдәҺ int.MaxValue

    вң… иҷҪз„¶е®һйҷ…дёӯжҲ‘д»¬еҫҲе°‘зӣҙжҺҘз”Ё int.MaxValue еҒҡдёҡеҠЎйҷҗеҲ¶пјҢдҪҶе®ғжҳҜдёҖдёӘйҮҚиҰҒзҡ„еҸӮиҖғеҖјгҖӮ

    вң… 3. зӣёе…іеёёйҮҸжӢ“еұ•

    йҷӨдәҶ intпјҢе…¶д»–ж•ҙж•°зұ»еһӢд№ҹжңүиҮӘе·ұзҡ„ MaxValue е’Ң MinValueпјҡ

    зұ»еһӢ жңҖеӨ§еҖјеӯ—ж®ө жңҖе°ҸеҖјеӯ—ж®ө иҢғеӣҙ
    int int.MaxValue int.MinValue -2,147,483,648 пҪһ 2,147,483,647
    long long.MaxValue long.MinValue -9,223,372,036,854,775,808 пҪһ 9,223,372,036,854,775,807
    short short.MaxValue short.MinValue -32,768 пҪһ 32,767
    byte byte.MaxValue byte.MinValue 0 пҪһ 255
    uint uint.MaxValue uint.MinValue 0 пҪһ 4,294,967,295
    ulong ulong.MaxValue ulong.MinValue 0 пҪһ 18,446,744,073,709,551,615

    рҹ§  е®ғд»¬зҡ„и®ҝй—®ж–№ејҸдёҖж ·пјҡlong.MaxValueгҖҒbyte.MinValue зӯү

    еӣӣгҖҒрҹ”’ жіЁж„ҸдәӢйЎ№

    жіЁж„ҸзӮ№ иҜҙжҳҺ
    вқҢ дёҚжҳҜж–№жі• int.MaxValue жҳҜеӯ—ж®өпјҢдёҚжҳҜеҮҪж•°пјҢдёҚиғҪеҠ жӢ¬еҸ·пјҢеҰӮ int.MaxValue() жҳҜй”ҷиҜҜзҡ„пјҒ
    вң… е…ЁеұҖжңүж•Ҳ иҝҷдәӣеёёйҮҸеңЁд»»дҪ• C# зЁӢеәҸдёӯйғҪеҸҜд»ҘзӣҙжҺҘдҪҝз”ЁпјҢеұһдәҺ System еҹәзЎҖзұ»еһӢзҡ„дёҖйғЁеҲҶ
    вң… зј–иҜ‘жңҹеёёйҮҸ е®ғд»¬зҡ„еҖјеңЁзј–иҜ‘ж—¶зЎ®е®ҡпјҢжҖ§иғҪй«ҳпјҢе®үе…ЁеҸҜйқ 
    вҡ пёҸ жәўеҮәйЈҺйҷ© и¶…еҮә int иҢғеӣҙзҡ„иҝҗз®—дёҚдјҡиҮӘеҠЁжҠҘй”ҷпјҢиҖҢжҳҜдә§з”ҹй”ҷиҜҜз»“жһңпјҲйҷӨйқһз”Ё checkedпјү
    вң… е®үе…Ёзј–зЁӢеҝ…еӨҮ еңЁеӨ„зҗҶз”ЁжҲ·иҫ“е…ҘгҖҒзҙҜеҠ и®Ўз®—гҖҒж•°з»„зҙўеј•гҖҒж•°еҖјжҜ”иҫғж—¶пјҢжҺЁиҚҗеҸӮиҖғиҝҷдәӣиҫ№з•ҢеҖј

    дә”гҖҒвң… жҖ»з»“пјҡint.MaxValue е’Ң int.MinValue зҡ„з”ЁйҖ”йҖҹжҹҘ

    з”ЁйҖ” иҜҙжҳҺ зӨәдҫӢ
    дәҶи§Јж•°жҚ®иҢғеӣҙ зҹҘйҒ“ int иғҪеӯҳеӨҡеӨ§ / еӨҡе°Ҹзҡ„ж•° int.MaxValue вҶ’ 2147483647
    йҳІжӯўжәўеҮә еҲӨж–ӯеҠ еҮҸд№ҳйҷӨжҳҜеҗҰи¶…еҮәиҢғеӣҙ if (a > int.MaxValue - b)
    еҲқе§ӢеҢ–жһҒеҖј жүҫжңҖеӨ§/жңҖе°ҸеҖјж—¶еҲқе§ӢеҢ–еҸҳйҮҸ int max = int.MinValue;
    иҫ“е…Ҙж ЎйӘҢ йҷҗеҲ¶ж•°еӯ—иҢғеӣҙпјҢйҒҝе…Қж— ж•Ҳиҫ“е…Ҙ жҜ”еҰӮе№ҙйҫ„дёҚиғҪе°ҸдәҺ 0 жҲ–еӨ§дәҺ 120
    зј–еҶҷеҒҘеЈ®д»Јз Ғ жҸҗеҚҮзЁӢеәҸзҡ„е®№й”ҷиғҪеҠӣдёҺе®үе…ЁжҖ§ еҜ№иҫ№з•ҢеҖјеҒҡеҲӨж–ӯпјҢйҒҝе…ҚејӮеёёиЎҢдёә

    вң… дёҖеҸҘиҜқжҖ»з»“пјҡ

    int.MaxValueпјҲ2147483647пјүе’Ң int.MinValueпјҲ-2147483648пјүжҳҜ C# дёӯиЎЁзӨә int зұ»еһӢиғҪеӯҳеӮЁзҡ„жңҖеӨ§е’ҢжңҖе°Ҹж•ҙж•°еҖјзҡ„дёӨдёӘйқҷжҖҒеёёйҮҸпјҲеӯ—ж®өпјүпјҢе®ғд»¬жҳҜиҝӣиЎҢж•°еҖјиҢғеӣҙеҲӨж–ӯгҖҒйҳІжӯўжәўеҮәгҖҒеҲқе§ӢеҢ–жңҖеҖјгҖҒе®үе…Ёзј–зЁӢзҡ„йҮҚиҰҒе·Ҙе…·пјҢеңЁе®һйҷ…ејҖеҸ‘дёӯе№ҝжіӣдҪҝз”ЁпјҢдҪҝз”Ёж—¶жіЁж„Ҹе®ғд»¬жҳҜеӯ—ж®өдёҚжҳҜж–№жі•пјҢжӯЈзЎ®еҶҷжі•дёә int.MaxValueпјҢдёҚиҰҒеҠ жӢ¬еҸ·гҖӮ

    еҰӮдҪ еёҢжңӣжҲ‘иҝӣдёҖжӯҘи®Іи§Јпјҡ

    • еҰӮдҪ•з”Ё checked е…ій”®еӯ—жЈҖжөӢиҝҗз®—жәўеҮә
    • еҰӮдҪ•е®үе…Ёи§Јжһҗз”ЁжҲ·иҫ“е…Ҙзҡ„ж•°еӯ—
    • е…¶д»–ж•ҙж•°/жө®зӮ№зұ»еһӢзҡ„ MaxValue / MinValue
    • е®һжҲҳжЎҲдҫӢпјҡз»ҹи®ЎжңҖеӨ§еҖјгҖҒзҙҜеҠ жұӮе’ҢйҳІжәўеҮәзӯү

    ж¬ўиҝҺз»§з»ӯжҸҗй—®пјҒжҲ‘еҸҜд»ҘдёәдҪ жҸҗдҫӣиҜҰз»Ҷд»Јз ҒдёҺжңҖдҪіе®һи·ө рҹҳҠгҖӮ
