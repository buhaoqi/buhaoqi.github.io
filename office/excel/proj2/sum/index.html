    2. жЁЎеқ—дәҢ Excelеә”з”Ё
    3. йЎ№зӣ®дәҢ е…¬ејҸдёҺеҮҪж•°

    SUMMеҮҪж•°

    ExcelжұӮе’ҢеҮҪж•°иҜҰи§ЈпјҡSUM()гҖҒSUMIF()гҖҒSUMIFS()

    дёҖгҖҒSUM() еҮҪж•°пјҡеҹәзЎҖжұӮе’Ң¶

    1. еҹәжң¬иҜӯжі•¶

    =SUM(number1, [number2], ...)

    2. еҸӮж•°иҜҙжҳҺ¶

    • number1пјҡеҝ…йңҖпјҢз¬¬дёҖдёӘж•°еӯ—гҖҒеҚ•е…ғж јеј•з”ЁжҲ–еҢәеҹҹ
    • number2, ...пјҡеҸҜйҖүпјҢжңҖеӨҡ255дёӘеҸӮж•°

    3. дҪҝз”ЁзӨәдҫӢ¶

    зӨәдҫӢ1пјҡзӣҙжҺҘж•°еҖјжұӮе’Ң¶

    =SUM(10, 20, 30)          // з»“жһңпјҡ60

    зӨәдҫӢ2пјҡеҚ•е…ғж јеј•з”ЁжұӮе’Ң¶

    |   A  |
|------|
|   10 |
|   20 |
|   30 |

=SUM(A1:A3)               // з»“жһңпјҡ60
=SUM(A1, A2, A3)          // з»“жһңпјҡ60

    зӨәдҫӢ3пјҡж··еҗҲеҸӮж•°¶

    |   A  |   B  |
|------|------|
|   10 |   5  |
|   20 |   8  |
|   30 |   3  |

=SUM(A1:A3, 40, B1:B2)    // з»“жһңпјҡ10+20+30+40+5+8 = 113

    зӨәдҫӢ4пјҡи·іиҝҮйқһж•°еҖј¶

    |   A      |
|----------|
|   10     |
|   ж–Үжң¬   |
|   20     |
|   TRUE   |

=SUM(A1:A4)               // з»“жһңпјҡ31пјҲ10+20+1пјҢTRUEиҪ¬дёә1пјҢж–Үжң¬иў«еҝҪз•Ҙпјү

    4. зү№ж®ҠеҠҹиғҪ¶

    |   A  |   B  |
|------|------|
|   2  |   5  |
|   3  |   4  |
|   4  |   3  |

=SUM(A1:A3*B1:B3)       // ж•°з»„е…¬ејҸпјҡ2*5 + 3*4 + 4*3 = 10+12+12 = 34

|   C  |
|------|
|   45 |
|   55 |
|   38 |
|   62 |

=SUM(--(C1:C4>50))      // з»ҹи®ЎеӨ§дәҺ50зҡ„дёӘж•°пјҡз»“жһңдёә2

    дәҢгҖҒSUMIF() еҮҪж•°пјҡеҚ•жқЎд»¶жұӮе’Ң¶

    1. еҹәжң¬иҜӯжі•¶

    =SUMIF(range, criteria, [sum_range])

    2. еҸӮж•°иҜҰи§Ј¶

    еҸӮж•° иҜҙжҳҺ еҝ…йңҖ
    range иҰҒеә”з”ЁжқЎд»¶зҡ„еҚ•е…ғж јеҢәеҹҹ жҳҜ
    criteria жқЎд»¶пјҢеҸҜд»ҘжҳҜж•°еӯ—гҖҒиЎЁиҫҫејҸгҖҒж–Үжң¬жҲ–еҚ•е…ғж јеј•з”Ё жҳҜ
    sum_range иҰҒжұӮе’Ңзҡ„е®һйҷ…еҚ•е…ғж јеҢәеҹҹпјҲеҸҜйҖүпјҢзңҒз•Ҙж—¶еҜ№rangeжұӮе’Ңпјү еҗҰ

    3. жқЎд»¶еҶҷжі•еӨ§е…Ё¶

    зӨәдҫӢж•°жҚ®иЎЁ1пјҡж°ҙжһңй”Җе”®¶

    A B
    иӢ№жһң 50
    йҰҷи•ү 30
    иӢ№жһң 40
    ж©ҷеӯҗ 25
    иӢ№жһңжұҒ 35
    жўЁ 45 
    =SUMIF(A:A,"иӢ№жһң",B:B)      // зӯүдәҺ"иӢ№жһң"пјҡ50+40 = 90
=SUMIF(A:A,"*жһң",B:B)       // д»Ҙ"жһң"з»“е°ҫпјҡ50+30+40+45 = 165
=SUMIF(A:A,"иӢ№*",B:B)       // д»Ҙ"иӢ№"ејҖеӨҙпјҡ50+40+35 = 125
=SUMIF(A:A,"?жһң",B:B)       // дёӨдёӘеӯ—дё”д»Ҙ"жһң"з»“е°ҫпјҡ50+30+40 = 120
=SUMIF(A:A,"<>иӢ№жһң",B:B)    // дёҚзӯүдәҺ"иӢ№жһң"пјҡ30+25+35+45 = 135

    зӨәдҫӢж•°жҚ®иЎЁ2пјҡй”Җе”®йҮ‘йўқ¶

    B
    150
    80
    220
    95
    310
    45 
    =SUMIF(B:B,">100")          // еӨ§дәҺ100пјҡ150+220+310 = 680
=SUMIF(B:B,">=500")         // еӨ§дәҺзӯүдәҺ500пјҡ0
=SUMIF(B:B,"<>0")           // дёҚзӯүдәҺ0пјҡ150+80+220+95+310+45 = 900
=SUMIF(B:B,"<1000")         // е°ҸдәҺ1000пјҡ900
=SUMIF(B:B,150)             // зӯүдәҺ150пјҡ150

    зӨәдҫӢж•°жҚ®иЎЁ3пјҡж—Ҙжңҹй”Җе”®¶

    C D
    2024-01-05 500
    2024-02-10 300
    2024-01-15 400
    2024-03-20 600 
    =SUMIF(C:C,">=2024-01-01",D:D)     // 500+300+400+600 = 1800
=SUMIF(C:C,">"&DATE(2024,1,10),D:D) // еӨ§дәҺ2024-01-10пјҡ300+400+600 = 1300
=SUMIF(C:C,">="&TODAY(),D:D)       // еӨ§дәҺзӯүдәҺд»ҠеӨ©

    еҚ•е…ғж јеј•з”ЁдҪңдёәжқЎд»¶¶

    |   D2  |   E2  |
|-------|-------|
| иӢ№жһң  |  100  |

=SUMIF(A:A,D2,B:B)          // D2зҡ„еҶ…е®№дҪңдёәжқЎд»¶
=SUMIF(B:B,">"&E2)          // еӨ§дәҺE2зҡ„еҖј

    4. е®һйҷ…еә”з”ЁзӨәдҫӢ¶

    зӨәдҫӢж•°жҚ®иЎЁ4пјҡй”Җе”®ж•°жҚ®¶

    A B
    й”Җе”®е‘ҳ й”Җе”®йўқ
    еј дёү 5000
    жқҺеӣӣ 8000
    еј дёү 3000
    зҺӢдә” 6000
    жқҺеӣӣ 4000
    еј дёү 2000 
    жұӮ"еј дёү"зҡ„й”Җе”®йўқпјҡ=SUMIF(A:A,"еј дёү",B:B)      // 5000+3000+2000 = 10000
жұӮй”Җе”®йўқ>5000зҡ„жҖ»е’Ңпјҡ=SUMIF(B:B,">5000")      // 8000+6000 = 14000
жұӮйқһ"еј дёү"зҡ„й”Җе”®йўқпјҡ=SUMIF(A:A,"<>еј дёү",B:B)  // 8000+6000+4000 = 18000

    зӨәдҫӢж•°жҚ®иЎЁ5пјҡеӯҰз”ҹжҲҗз»©¶

    A B
    зҸӯзә§ жҲҗз»©
    1зҸӯ 85
    2зҸӯ 78
    1зҸӯ 92
    3зҸӯ 45
    1зҸӯ 88
    2зҸӯ 65
    жқҺеҚҺ 90
    жқҺжҳҺ 75 
    жұӮ1зҸӯжҖ»жҲҗз»©пјҡ=SUMIF(A:A,"1зҸӯ",B:B)           // 85+92+88 = 265
жұӮеҸҠж ј(>=60)жҖ»жҲҗз»©пјҡ=SUMIF(B:B,">=60")       // 85+78+92+88+65+90+75 = 573
жұӮе§“"жқҺ"зҡ„еӯҰз”ҹжҲҗз»©пјҡ=SUMIF(A:A,"жқҺ*",B:B)    // 90+75 = 165

    5. еёёи§Ғй”ҷиҜҜеҸҠеӨ„зҗҶ¶

    D2 B
    100 150
    80
    220 
    й”ҷиҜҜеҶҷжі•пјҡ=SUMIF(B:B,">D2")       // D2дјҡиў«еҪ“дҪңж–Үжң¬"100"
жӯЈзЎ®еҶҷжі•пјҡ=SUMIF(B:B,">"&D2)     // з”Ё&иҝһжҺҘпјҢзҗҶи§Јдёә">100"

й”ҷиҜҜпјҡ=SUMIF(A:A,жқЎд»¶,B1:B10)     // еҢәеҹҹеӨ§е°ҸдёҚдёҖиҮҙ
жӯЈзЎ®пјҡ=SUMIF(A1:A10,жқЎд»¶,B1:B10)  // еҢәеҹҹеӨ§е°ҸзӣёеҗҢ

    дёүгҖҒSUMIFS() еҮҪж•°пјҡеӨҡжқЎд»¶жұӮе’Ң¶

    1. еҹәжң¬иҜӯжі•¶

    =SUMIFS(sum_range, criteria_range1, criteria1, 
        [criteria_range2, criteria2], ...)

    2. еҸӮж•°иҜҙжҳҺ¶

    еҸӮж•° иҜҙжҳҺ
    sum_range иҰҒжұӮе’Ңзҡ„е®һйҷ…еҚ•е…ғж јеҢәеҹҹ
    criteria_range1 з¬¬дёҖдёӘжқЎд»¶зҡ„еҢәеҹҹ
    criteria1 з¬¬дёҖдёӘжқЎд»¶
    criteria_range2, criteria2, ... еҸҜйҖүпјҢжңҖеӨҡ127дёӘжқЎд»¶еҜ№

    3. дҪҝз”ЁзӨәдҫӢ¶

    зӨәдҫӢж•°жҚ®иЎЁ6пјҡй”Җе”®жҳҺз»Ҷ¶

    A B C
    йғЁй—Ё жңҲд»Ҫ й”Җе”®йўқ
    й”Җе”®йғЁ 1жңҲ 5000
    жҠҖжңҜйғЁ 1жңҲ 3000
    й”Җе”®йғЁ 2жңҲ 6000
    дәәдәӢйғЁ 1жңҲ 2000
    й”Җе”®йғЁ 1жңҲ 4000
    жҠҖжңҜйғЁ 2жңҲ 3500 
    жұӮ"й”Җе”®йғЁ"еңЁ"1жңҲ"зҡ„й”Җе”®йўқпјҡ
=SUMIFS(C:C, A:A,"й”Җе”®йғЁ", B:B,"1жңҲ")  // 5000+4000 = 9000

жұӮ"й”Җе”®йғЁ"еңЁ"1жңҲ"дё”й”Җе”®йўқ>4000зҡ„и®°еҪ•пјҡ
=SUMIFS(C:C, A:A,"й”Җе”®йғЁ", B:B,"1жңҲ", C:C,">4000")  // 5000

жұӮ1жңҲд»ҪжүҖжңүйғЁй—Ёй”Җе”®йўқпјҡ
=SUMIFS(C:C, B:B,"1жңҲ")  // 5000+3000+2000+4000 = 14000

    зӨәдҫӢж•°жҚ®иЎЁ7пјҡж—ҘжңҹиҢғеӣҙй”Җе”®¶

    A B
    ж—Ҙжңҹ й”Җе”®йўқ
    2024-01-05 500
    2024-01-15 300
    2024-02-10 400
    2024-01-25 600
    2024-02-20 200 
    жұӮ2024е№ҙ1жңҲзҡ„й”Җе”®йўқпјҡ
=SUMIFS(B:B, A:A,">=2024-01-01", A:A,"<=2024-01-31")  // 500+300+600 = 1400

    4. е®һйҷ…е·ҘдҪңеңәжҷҜ¶

    еңәжҷҜ1пјҡй”Җе”®ж•°жҚ®еҲҶжһҗ¶

    й”Җе”®е‘ҳ дә§е“Ғ ең°еҢә й”Җе”®йўқ ж—Ҙжңҹ
    еј дёү жүӢжңә еҚҺдёң 5000 2024-01-05
    жқҺеӣӣ з”өи„‘ еҚҺеҚ— 8000 2024-01-10
    еј дёү жүӢжңә еҚҺеҢ— 6000 2024-02-15
    зҺӢдә” е№іжқҝ еҚҺдёң 3000 2024-01-20
    еј дёү з”өи„‘ еҚҺдёң 4000 2024-02-25 
    1. еј дёүеңЁеҚҺдёңеҢәй”Җе”®жүӢжңәзҡ„жҖ»йҮ‘йўқпјҡ
   =SUMIFS(й”Җе”®йўқеҲ—, й”Җе”®е‘ҳеҲ—,"еј дёү", дә§е“ҒеҲ—,"жүӢжңә", ең°еҢәеҲ—,"еҚҺдёң")  // 5000

2. 2024е№ҙQ1пјҲ1-3жңҲпјүжүӢжңәй”Җе”®йўқпјҡ
   =SUMIFS(й”Җе”®йўқеҲ—, дә§е“ҒеҲ—,"жүӢжңә", ж—ҘжңҹеҲ—,">=2024-01-01", ж—ҘжңҹеҲ—,"<=2024-03-31")  // 5000+6000 = 11000

3. й”Җе”®йўқ>4000дё”дә§е“ҒдёәжүӢжңәжҲ–з”өи„‘зҡ„и®°еҪ•пјҡ
   йңҖиҰҒз»“еҗҲеӨҡдёӘSUMIFSпјҡ
   =SUMIFS(й”Җе”®йўқеҲ—, й”Җе”®йўқеҲ—,">4000", дә§е“ҒеҲ—,"жүӢжңә") + SUMIFS(й”Җе”®йўқеҲ—, й”Җе”®йўқеҲ—,">4000", дә§е“ҒеҲ—,"з”өи„‘")  // 5000+8000+6000 = 19000
